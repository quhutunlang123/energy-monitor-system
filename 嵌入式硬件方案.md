# 能源监测平台模拟设备实现方案

## 1. 模拟设备方案概述

本方案针对家庭能源监测平台设计，主要监测电力、燃气和水资源的使用情况。由于用户主要专注于软件开发，本方案采用纯软件模拟实现方式，无需实际硬件设备：
- **模拟数据方案**：通过软件模拟传感器数据，实现完整的能源监测功能
- **本地部署**：完全在用户本地环境运行，无需云服务，零成本实现

### 3.1 数据生成算法设计

#### 3.1.1 电力数据生成

```java
public class ElectricityDataGenerator {
    private Random random = new Random();
    private Map<String, Double> deviceState = new HashMap<>();
    
    public Map<String, Object> generateData(String deviceId, DeviceConfig config) {
        // 基础参数
        double voltage = 210 + random.nextDouble() * 20;
        double baseCurrent = config.getBaseCurrent();
        double currentVariation = config.getCurrentVariation();
        
        // 模拟设备开关状态
        if (!deviceState.containsKey(deviceId)) {
            deviceState.put(deviceId, 0.0);
        }
        
        // 模拟用电高峰和低谷
        int hour = LocalDateTime.now().getHour();
        double timeFactor = 1.0;
        if (hour >= 7 && hour <= 9) timeFactor = 1.5; // 早高峰
        if (hour >= 17 && hour <= 21) timeFactor = 1.8; // 晚高峰
        if (hour >= 0 && hour <= 5) timeFactor = 0.6; // 低谷
        
        // 计算电流
        double current = baseCurrent + random.nextDouble() * currentVariation * timeFactor;
        
        // 模拟设备启停
        if (random.nextDouble() < 0.05) { // 5%概率改变状态
            if (deviceState.get(deviceId) > 0) {
                current = 0.0;
                deviceState.put(deviceId, 0.0);
            } else {
                current = baseCurrent + random.nextDouble() * currentVariation * timeFactor;
                deviceState.put(deviceId, current);
            }
        }
        
        // 计算功率和用电量
        double power = voltage * current;
        double energy = power * 5.0 / 3600.0; // 5秒用电量(kWh)
        
        Map<String, Object> data = new HashMap<>();
        data.put("deviceId", deviceId);
        data.put("type", "electricity");
        data.put("voltage", Math.round(voltage * 100) / 100.0);
        data.put("current", Math.round(current * 100) / 100.0);
        data.put("power", Math.round(power * 100) / 100.0);
        data.put("energy", Math.round(energy * 1000) / 1000.0);
        data.put("timestamp", System.currentTimeMillis());
        
        return data;
    }
}
```

public class GasDataGenerator {
    private Random random = new Random();
    private Map<String, Boolean> deviceState = new HashMap<>();
    
    public Map<String, Object> generateData(String deviceId, DeviceConfig config) {
        // 基础参数
        double baseFlowRate = config.getBaseFlowRate();
        double flowVariation = config.getFlowVariation();
        double basePressure = 0.02;
        
        // 模拟设备开关状态
        if (!deviceState.containsKey(deviceId)) {
            deviceState.put(deviceId, false);
        }
        
        // 模拟用气时段
        int hour = LocalDateTime.now().getHour();
        double timeFactor = 1.0;
        if (hour >= 7 && hour <= 8) timeFactor = 1.2; // 早餐
        if (hour >= 12 && hour <= 13) timeFactor = 1.3; // 午餐
        if (hour >= 18 && hour <= 19) timeFactor = 1.5; // 晚餐
        if (hour >= 0 && hour <= 6) timeFactor = 0.2; // 夜间
        
        // 模拟设备启停
        double flowRate = 0.0;
        if (random.nextDouble() < 0.03) { // 3%概率改变状态
            deviceState.put(deviceId, !deviceState.get(deviceId));
        }
        
        if (deviceState.get(deviceId)) {
            flowRate = baseFlowRate + random.nextDouble() * flowVariation * timeFactor;
        }
        
        // 压力与流量相关
        double pressure = basePressure + flowRate * 0.001;
        
        Map<String, Object> data = new HashMap<>();
        data.put("deviceId", deviceId);
        data.put("type", "gas");
        data.put("flowRate", Math.round(flowRate * 100) / 100.0);
        data.put("pressure", Math.round(pressure * 1000) / 1000.0);
        data.put("timestamp", System.currentTimeMillis());
        
        return data;
    }
}

#### 3.1.3 水资源数据生成

```java
public class WaterDataGenerator {
    private Random random = new Random();
    private Map<String, Boolean> deviceState = new HashMap<>();
    
    public Map<String, Object> generateData(String deviceId, DeviceConfig config) {
        // 基础参数
        double baseFlowRate = config.getBaseFlowRate();
        double flowVariation = config.getFlowVariation();
        double baseWaterLevel = 150;
        
        // 模拟设备开关状态
        if (!deviceState.containsKey(deviceId)) {
            deviceState.put(deviceId, false);
        }
        
        // 模拟用水时段
        int hour = LocalDateTime.now().getHour();
        double timeFactor = 1.0;
        if (hour >= 7 && hour <= 8) timeFactor = 1.4; // 早高峰
        if (hour >= 12 && hour <= 13) timeFactor = 1.2; // 午间
        if (hour >= 18 && hour <= 20) timeFactor = 1.6; // 晚高峰
        if (hour >= 0 && hour <= 5) timeFactor = 0.3; // 夜间
        
        // 模拟设备启停
        double flowRate = 0.0;
        if (random.nextDouble() < 0.04) { // 4%概率改变状态
            deviceState.put(deviceId, !deviceState.get(deviceId));
        }
        
        if (deviceState.get(deviceId)) {
            flowRate = baseFlowRate + random.nextDouble() * flowVariation * timeFactor;
        }
        
        // 水位与用水量相关
        double waterLevel = baseWaterLevel - flowRate * 0.1;
        if (waterLevel < 50) waterLevel = 50;
        
        // 水质参数
        double phValue = 6.8 + random.nextDouble() * 0.6;
        double turbidity = 5 + random.nextDouble() * 20;
        
        Map<String, Object> data = new HashMap<>();
        data.put("deviceId", deviceId);
        data.put("type", "water");
        data.put("flowRate", Math.round(flowRate * 100) / 100.0);
        data.put("waterLevel", Math.round(waterLevel * 10) / 10.0);
        data.put("phValue", Math.round(phValue * 10) / 10.0);
        data.put("turbidity", Math.round(turbidity * 10) / 10.0);
        data.put("timestamp", System.currentTimeMillis());
        
        return data;
    }
}
```

## 4. 模拟数据传输方案

### 4.1 MQTT消息传输

```java
@Component
public class MqttDataPublisher {
    private MqttClient client;
    private String topic = "energy/monitor";
    
    @PostConstruct
    public void init() {
        try {
            client = new MqttClient("tcp://localhost:1883", "SimulatorClient", new MemoryPersistence());
            MqttConnectOptions options = new MqttConnectOptions();
            options.setCleanSession(true);
            client.connect(options);
        } catch (MqttException e) {
            e.printStackTrace();
        }
    }
    
    public void publish(Map<String, Object> data) {
        try {
            String payload = new ObjectMapper().writeValueAsString(data);
            MqttMessage message = new MqttMessage(payload.getBytes());
            message.setQos(1);
            client.publish(topic, message);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
    
    @PreDestroy
    public void destroy() {
        try {
            if (client != null && client.isConnected()) {
                client.disconnect();
            }
        } catch (MqttException e) {
            e.printStackTrace();
        }
    }
}
```

### 4.2 HTTP数据传输

```java
@Component
public class HttpDataPublisher {
    private RestTemplate restTemplate;
    private String apiUrl = "http://localhost:8080/api/data";
    
    @Autowired
    public HttpDataPublisher(RestTemplate restTemplate) {
        this.restTemplate = restTemplate;
    }
    
    public void publish(Map<String, Object> data) {
        try {
            ResponseEntity<String> response = restTemplate.postForEntity(
                apiUrl, data, String.class
            );
            if (!response.getStatusCode().is2xxSuccessful()) {
                System.err.println("Failed to send data: " + response.getStatusCode());
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
```

## 9. 性能优化策略

### 9.1 数据生成优化

- **批量生成**：批量生成数据，减少计算开销
- **缓存机制**：缓存设备状态，避免重复计算
- **并行处理**：多线程并行生成不同设备的数据
- **自适应频率**：根据系统负载调整数据生成频率

### 9.2 传输优化

- **批量传输**：批量传输数据，减少网络请求
- **压缩传输**：压缩数据 payload，减少网络带宽
- **失败重试**：实现数据传输失败重试机制
- **负载均衡**：多后端服务实例时实现负载均衡

### 9.3 资源优化

- **内存管理**：合理管理内存，避免内存泄漏
- **线程池**：使用线程池管理并发任务
- **连接池**：使用连接池管理MQTT和HTTP连接
- **资源限制**：设置合理的CPU和内存限制

### 6.1 模拟数据生成器

创建一个独立的服务，模拟传感器数据并通过MQTT或HTTP发送给后端服务。

#### 6.1.1 Python模拟数据生成器

```python
import time
import random
import paho.mqtt.client as mqtt
import json

# MQTT配置
MQTT_BROKER = "localhost"
MQTT_PORT = 1883
MQTT_TOPIC = "energy/monitor"

# 初始化MQTT客户端
client = mqtt.Client()
client.connect(MQTT_BROKER, MQTT_PORT, 60)

def generate_electricity_data():
    """生成电力模拟数据"""
    return {
        "type": "electricity",
        "voltage": round(random.uniform(210, 230), 2),
        "current": round(random.uniform(0.5, 10), 2),
        "power": round(random.uniform(100, 2000), 2),
        "energy": round(random.uniform(0, 1), 3),
        "timestamp": int(time.time() * 1000)
    }

def generate_gas_data():
    """生成燃气模拟数据"""
    return {
        "type": "gas",
        "flowRate": round(random.uniform(0, 5), 2),
        "pressure": round(random.uniform(0.02, 0.05), 3),
        "timestamp": int(time.time() * 1000)
    }

def generate_water_data():
    """生成水资源模拟数据"""
    return {
        "type": "water",
        "flowRate": round(random.uniform(0, 10), 2),
        "waterLevel": round(random.uniform(50, 200), 1),
        "phValue": round(random.uniform(6.5, 7.5), 1),
        "turbidity": round(random.uniform(0, 100), 1),
        "timestamp": int(time.time() * 1000)
    }

try:
    while True:
        # 生成并发送电力数据
        electricity_data = generate_electricity_data()
        client.publish(MQTT_TOPIC, json.dumps(electricity_data))
        print("Electricity:", electricity_data)
        
        # 生成并发送燃气数据
        gas_data = generate_gas_data()
        client.publish(MQTT_TOPIC, json.dumps(gas_data))
        print("Gas:", gas_data)
        
        # 生成并发送水资源数据
        water_data = generate_water_data()
        client.publish(MQTT_TOPIC, json.dumps(water_data))
        print("Water:", water_data)
        
        time.sleep(5)  # 5秒发送一次数据
except KeyboardInterrupt:
    client.disconnect()
```
